namespace = jizya_events

country_event = {
	id = jizya_events.1
	title = jizya_events.1.t
	desc = jizya_events.1.d
	picture = PRAYING_eventPicture
	
	is_triggered_only = yes
	
	trigger = {
		NOT = { 
			has_country_flag = jizya_yes
			has_country_flag = jizya_no
		}
		OR = {
			religion = sunni
			religion = shiite
			religion = ibadi
		}
		technology_group = indian
		any_owned_province = {
			religion = hinduism
		}	
	}
	mean_time_to_happen = {
		days = 1
	}
	
	option = {
		name = jizya_events.1.a
		ai_chance = {
			factor = 75
			modifier = {
				factor = 0
				is_at_war = yes
			}
			modifier = {
				factor = 0
				has_any_disaster = yes
			}	
		}
		set_country_flag = jizya_yes
		every_owned_province = {
			if = {
				limit = { religion = hinduism }
				add_unrest = 5
				add_province_modifier = {
					name = jizya_tax
					duration = 9125
				}
			}
		}	
	}
	option = {
		name = jizya_events.1.b
		set_country_flag = jizya_no
		ROOT = {
			every_owned_province = {
				limit = { religion = hinduism }
				add_unrest = -10
			}
		}	
		add_country_modifier = {
			name = no_jizya_tolerance
			duration = -1
		}	
	}	
}
